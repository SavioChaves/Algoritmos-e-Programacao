qtd_num = int(input())
soma_num = 0

if qtd_num <= 0:
    print("Informe uma quantidade > 0!")
else:
    while qtd_num > 0:
        dig_num = float(input())
        soma_num = soma_num + dig_num
        qtd_num = qtd_num - 1
    print("Soma dos n√∫meros informados: %.2f" %(soma_num))
